def findMedianSortedArrays(nums1, nums2):
    # Ensure nums1 is the shorter array
    if len(nums1) > len(nums2):
        nums1, nums2 = nums2, nums1

    m, n = len(nums1), len(nums2)
    half = (m + n) # total elements for left half

    lo, hi = 0, m  # binary search bounds on nums1

    while lo <= hi:
        i = (lo + hi) // 2   # partition index in nums1
        j = (m + n + 1) // 2 - i  # derived partition in nums2

        # Edge values using Â±infinity for out-of-bounds
        maxLeft1  = nums1[i-1] if i > 0 else float('-inf')
        minRight1 = nums1[i]   if i < m else float('inf')
        maxLeft2  = nums2[j-1] if j > 0 else float('-inf')
        minRight2 = nums2[j]   if j < n else float('inf')

        if maxLeft1 <= minRight2 and maxLeft2 <= minRight1:
            # âœ“ Perfect partition found!
            if (m + n) % 2 == 1:
                return max(maxLeft1, maxLeft2)
            return (max(maxLeft1, maxLeft2) + min(minRight1, minRight2)) / 2

        elif maxLeft1 > minRight2:
            hi = i - 1  # i too large, move left
        else:
            lo = i + 1  # i too small, move right
